var gpu = new GPU();

var opt = {
    dimensions: [800, 600],
    debug: true,
    graphical: true,
    mode: 'gpu'
};

var A = [0, 1, 2];

var render = gpu.createKernel(function(A) {
    var x = 0;
    for (var i=0; i<A[2]; i++) {
        x++;
    }
    
    var y = 0;
    while (y < 5) {
        y++;
    }
    
    function test(x) {
        return x*x + A[0];
    }
    
    if (test(this.thread.x) % 20 == Math.floor(Math.sin(this.thread.y))) {
        this.color(0,0,0);
    } else if (this.thread.x % 20 == (20-this.thread.y % 20)) {
        this.color(1,0,0);
    } else if (this.thread.x % 20 == 0) {
        this.color(0,1,0);
    } else if (this.thread.y % 20 == 0) {
        this.color(0,0,1);
    } else {
        this.color(1,1,1);
    }
}, opt);

render(A);

var canvas = render.getCanvas();
document.getElementsByTagName('body')[0].appendChild(canvas);
