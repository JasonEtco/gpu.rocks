Matrix Multiplication

var gpu = new GPU();

var opt = {
    dimensions: [512, 512]
}

var mat_mult = gpu.createKernel(function(A, B) {
    var sum = 0;
    for (var i=0; i<512; i++) {
        sum += A[this.thread.y][i] * B[i][this.thread.x];
    }
    return sum;
}, opt);

var C = mat_mult(A, B);
